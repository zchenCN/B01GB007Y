    \begin{problem}{1}
    写出下列复数的实部、虚部、模和辐角：
        \begin{itemize}
            \item[(1)] $w = e^z$
            \item[(2)] $w = 1 - \cos\alpha + i\sin\alpha, 0 \leq \alpha \leq 2\pi$
            \item[(3)] $w = \frac{1-2i}{3-4i} - \frac{2-i}{5i}$
        \end{itemize}
    \end{problem}
    \begin{solution}
        \begin{itemize}
            \item[(1)] $w = e^z = e^{(x+iy)}=e^xe^{iy}$, $Re(w) = e^x\cos y, Im(w)=e^x\sin y,$ $r=e^x, \theta = y+2k\pi, k\in \mathbb{Z}$
            \item[(2)] $Re(w) = 1-\cos\alpha, Im(w) = \sin\alpha$\\ $r = \sqrt{(1-\cos\alpha)^2+\sin^2\alpha}=\sqrt{2(1-\cos\alpha)}=2\sin\frac{\alpha}{2}$\\$\tan\theta = \frac{\sin\alpha}{1-\cos\alpha} = \frac{2\sin\frac{\alpha}{2}\cos\frac{\alpha}{2}}{2\sin^2\frac{\alpha}{2}}=\cot\frac{\alpha}{2}=\tan(\frac{\pi}{2}-\frac{\alpha}{2})$, $\theta = \frac{\pi}{2}-\frac{\alpha}{2} +2k\pi, k\in \mathbb{Z} $
            \item[(3)]
                \begin{align}
                    \frac{1-2i}{3-4i} - \frac{2-i}{5i}&= \frac{1-2i}{3-4i}\frac{3+4i}{3+4i}-\frac{2-i}{5i}\frac{-5i}{-5i}\\
                    &= \frac{11-2i}{25}-\frac{-5-10i}{25}\\
                    &= \frac{16+8i}{25}
                \end{align}
                $Re(w) = \frac{16}{25},Im(w)=\frac{8}{25}, r = \frac{8\sqrt{5}}{25}, \theta =\arctan\frac{1}{2}+2k\pi, k\in \mathbb{Z} $ 
        \end{itemize}
    \end{solution}
    
    \begin{problem}{2}
        求下列序列$\{z_n\}$的聚点和极限：
        \begin{itemize}
            \item[(1)] $z_n = (1+\frac{i}{n})\sin\frac{n\pi}{6}$ 
            \item[(2)] $z_n = (1+\frac{1}{2n})\cos\frac{n\pi}{3}$
        \end{itemize}
    \end{problem}
    \begin{solution}
        \begin{itemize}
            \item[(1)] 令$n_k=12k + l, l=0, 1,\cdots, 11$, 当$l=0, z_{n_k} = 0$; 当$l=1, z_{n_k} = \frac{1}{2} + i\frac{1}{24k+2}\rightarrow  \frac{1}{2}$ ,同样讨论可得其他聚点分别是$0, \pm\frac{1}{2},\pm\frac{\sqrt{3}}{2}, \pm1$
            \item[(2)] 令$n_k=6k+l, l=0, 1, \cdots, 5$, 当$l=0$时, $z_{n_k} = 1+\frac{1}{12k+2}\rightarrow 1$同样讨论可得其他聚点分别是$ \pm\frac{\sqrt{3}}{2}, \pm1$
        \end{itemize}
    \end{solution}
    
    \begin{problem}{3}
        计算$(1+i)^n+(1-i)^n, n\in\mathbb{Z}$
    \end{problem}
    \begin{solution}:一般地我们有
        \begin{equation*}
            (\cos\theta+i\sin\theta)^n = (e^{i\theta})^n = e^{in\theta} = \cos n\theta+i\sin n\theta        
        \end{equation*}
        所以：
        \begin{align*} 
            (1+i)^n+(1-i)^n &= \sqrt{2}^n\left(\cos\frac{\pi}{4}+i\sin\frac{\pi}{4}\right)^n + \sqrt{2}^n\left(\cos\frac{-\pi}{4}+i\sin\frac{-\pi}{4}\right)^n \\
            &= \sqrt{2}^n\left(\cos\frac{n\pi}{4}+i\sin\frac{n\pi}{4}\right) + \sqrt{2}^n\left(\cos\frac{-n\pi}{4}+i\sin\frac{-n\pi}{4}\right) \\
            &= \sqrt{2}^n\left(\cos\frac{n\pi}{4}+i\sin\frac{n\pi}{4}\right) + \sqrt{2}^n\left(\cos\frac{n\pi}{4}-i\sin\frac{n\pi}{4}\right) \\
            &= 2\sqrt{2}^n\cos\frac{n\pi}{4}
        \end{align*}
    \end{solution}
