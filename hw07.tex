    \begin{problem}{1}
        将下列函数在指定点展开为Taylor级数，并给出其收敛半径
        \begin{itemize}
            \item[(1)] ~$1-z^2$~在~$z=1$~展开
            \item[(2)] ~$\sin z$~在~$z=n\pi$~展开
            \item[(3)] ~$\frac{1}{1+z+z^2}$~在~$z=0$~展开
            \item[(4)] ~$\frac{z^2}{(z+1)^2}$~在~$z=1$~展开
        \end{itemize}
    \end{problem}
    \begin{solution}
        \begin{itemize}
            \item[(1)] ~$1 - z^2 = -2(z-1) - (z-1)^2, |z| < \infty$~
            \item[(2)] 
                \begin{align*}
                     \sin z &= \sin(z-n\pi + n\pi)\\
                     &= \sin(z-n\pi)\cos n\pi + \cos(z-n\pi)\sin n\pi\\
                     &= (-1)^n\sum\limits_{k=0}^{\infty}\frac{(-1)^k(z-n\pi)^{2k+1}}{(2k+1)!}\\
                     &= \sum\limits_{k=0}^{\infty}\frac{(-1)^{k+n}(z-n\pi)^{2k+1}}{(2k+1)!}, |z| < \infty
                \end{align*}
            \item[(3)]
                \begin{align*}
                    \frac{1}{1+z+z^2} &= \frac{(1-z)}{(1-z)(1+z+z^2)}\\
                    &= \frac{1-z}{1-z^3}\\
                    &= (1-z)\sum_{n=0}^{\infty}(z^3)^n\\
                    &= \sum_{n=0}^{\infty}z^{3n} - \sum_{n=0}^{\infty}z^{3n+1}\\
                    &= \sum_{n=0}^{\infty}\left(z^{3n}-z^{3n+1}\right), |z|<1
                \end{align*}
            \item[(4)]
            \begin{align*}
                \frac{z^2}{(z+1)^2} &= -z^2\left(\frac{1}{z+1}\right)^{'}\\
                &= -\left[(z-1)^2+2(z-1)+1\right]\left(\frac{1}{2}\frac{1}{1+\frac{z-1}{2}}\right)^{'}\\
                &= -\left[(z-1)^2+2(z-1)+1\right]\left(\sum_{n=0}^{\infty}\frac{(z-1)^n}{2^{n+1}}\right)^{'}\\
                &= - \sum_{n=1}^{\infty}\frac{n(z-1)^{n-1}}{2^{n+1}}\left[(z-1)^2+2(z-1)+1\right], |z-1|<2
            \end{align*}
            \begin{align*}
                \frac{z^2}{(z+1)^2} &= 1 - \frac{2}{z+1} + \left(\frac{1}{z+1}\right)^2\\
                &= 1 - \frac{2}{z+1} - \left(\frac{1}{z+1}\right)^{'}\\
                &= 1 - \frac{1}{1 + \frac{z-1}{2}} - \left(\frac{1}{2}\frac{1}{1+ \frac{z-1}{2}}\right)^{'}\\
                &= 1 - \sum_{n=0}^{\infty}\left(\frac{z-1}{2}\right)^n - \left(\sum_{n=0}^{\infty}\frac{(z-1)^n}{2^{n+1}}\right)^{'}\\
                &= 1 - \sum_{n=0}^{\infty}\left(\frac{z-1}{2}\right)^n - \sum_{n=1}^{\infty}\frac{n(z-1)^{n-1}}{2^{n+1}}
            \end{align*}
        \end{itemize}
    \end{solution}


    \begin{problem}{2}
        将下列函数在指定点展开为Taylor级数，并给出其收敛半径
        \begin{itemize}
            \item[(1)] ~$Ln(z)$~, 在~$z=i$~展开, 规定~$0\leq \arg < 2\pi$~ 
            \item[(2)] ~$Ln(z)$~, 在~$z=i$~展开, 规定~$Ln(z)\bigg|_{z=i}=-\frac{3}{2}\pi i$~
        \end{itemize}
    \end{problem}
    \begin{solution}
    有~$Ln(z) =\sum_{n=0}^{\infty} \frac{(Ln(a))^{(n)}}{n!}(z-a)^n, (Ln(z))^{(n)} = (-1)^{n+1}\frac{(n-1)!}{z^n}, n=1, 2, \cdots$~
    \begin{itemize}
        \item[(1)] 可知~$Ln(i) = \frac{\pi i}{2}$~, 所以
        \begin{equation*}
            Ln(z) = \frac{\pi i}{2} - \sum_{n=1}^{\infty}\frac{i^n}{n}(z-i)^n
        \end{equation*}
        \item[(2)] 同理, 可知~$Ln(i) = -\frac{3\pi i}{2}$~, 所以
        \begin{equation*}
            Ln(z) = -\frac{3\pi i}{2} - \sum_{n=1}^{\infty}\frac{i^n}{n}(z-i)^n
        \end{equation*}
    \end{itemize}
    \end{solution}
    
    \begin{problem}{3}
        求下列函数的Laurent展开
        \begin{itemize}
            \item[(1)] ~$\frac{1}{z^2 - 3z + 2}$~, 展开区域为~$1<|z|<2$~
            \item[(3)] ~$\frac{1}{z^2 - 3z + 2}$~, 展开区域为~$2<|z|<\infty$~
        \end{itemize}
    \end{problem}
    \begin{solution}
    \begin{itemize}
        \item[(1)] 
        \begin{align*}
            \frac{1}{z^2 - 3z + 2} &= \frac{1}{(z-1)(z-2)} = \frac{1}{z-2}-\frac{1}{z-1}\\
            &= -\frac{1}{2}\frac{1}{1-\frac{z}{2}} - \frac{1}{z}\frac{1}{1-\frac{1}{z}}\\
            &= -\frac{1}{2}\sum_{n=0}^{\infty}\frac{z^n}{2^n} - \frac{1}{z}\sum_{n=0}^{\infty}\frac{1}{z^n}\\
            &= -\sum_{n=0}^{\infty}\frac{z^n}{2^{n+1}} - \sum_{n=0}^{\infty}\frac{1}{z^n}
        \end{align*}
        \item[(2)] 
        \begin{align*}
            \frac{1}{z^2 - 3z + 2} &= \frac{1}{(z-1)(z-2)} = \frac{1}{z-2}-\frac{1}{z-1}\\
            &= \frac{1}{z}\frac{1}{1-\frac{2}{z}} - \frac{1}{z}\frac{1}{1-\frac{1}{z}}\\
            &= \frac{1}{z}\sum_{n=0}^{\infty}\frac{2^n}{z^n} - \frac{1}{z}\sum_{n=0}^{\infty}\frac{1}{z^n}\\
            &= \sum_{n=0}^{\infty}\frac{2^n-1}{z^{n+1}}\\
            &= \sum_{n=2}^{\infty}\frac{2^{n-1}-1}{z^n}
        \end{align*}
    \end{itemize}
    \end{solution}
