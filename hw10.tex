    \begin{problem}{1}
        证明
        $$ f_1(z) = 1 + az + a^2z^2 + a^3z^3 + \cdots$$
        与
        $$ f_2(z) = \frac{1}{1 - z} - \frac{(1-a)z}{(1-z)^2} + \frac{(1-a)^2z^2}{(1-z)^3}$$
        互为解析延拓
    \end{problem}
    \begin{solution}
        显然有$$f_1(z) = \lim_{n\to\infty}\frac{1-(az)^{n+1}}{1-az} = \frac{1}{1-az}, \quad |z| < \frac{1}{|a|}$$观察发现
        \begin{equation*}
        \begin{split}
        (1-z)f_2(z) &= 1 + \left(-\frac{(1-a)z}{1-z}\right) + \left(-\frac{(1-a)z}{1-z}\right)^2 + \cdots\\
        &= \lim_{n\to\infty}\frac{1 - \left(-\frac{(1-a)z}{1-z}\right)^{n+1}}{1 + \frac{(1-a)z}{1-z}}\\
        &= \frac{1 - z}{1 - az}, \quad \left|\frac{(1-a)z}{1-z}\right| < 1
        \end{split}
        \end{equation*}
        所以在两个收敛区域的交集部分(仍然是区域)有:
        $$ f_1(z) = f_2(z) = \frac{1}{1-az}$$
        它们互为解析延拓.
    \end{solution}
    
    \begin{problem}{2}
        证明级数
        $$ \sum_{n=1}^{\infty}\left(\frac{1}{1-z^{n+1}}  - \frac{1}{1 - z^n}\right)$$
        在区域$|z|<1$和$|z| > 1$内分别代表两个解析函数，但是不互为解析延拓
    \end{problem}
    \begin{solution}
        \begin{equation*}
            \begin{split}
                f(z) &= \sum_{n=1}^{\infty}\left(\frac{1}{1-z^{n+1}}  - \frac{1}{1 - z^n}\right)\\
                &= \frac{1}{1-z^2} - \frac{1}{1-z} + \frac{1}{1-z^3} - \frac{1}{1-z^2} + \cdots \\
                &= \lim_{n\to\infty}-\frac{1}{1-z} + \frac{1}{1-z^{n+1}}\\
                &= \begin{cases}
                \frac{-z}{1-z}, |z| < 1\\
                \frac{-1}{1-z}, |z| > 1
                \end{cases}
            \end{split}
        \end{equation*}
        即$f(z)$在区域$|z|<1$和$|z| > 1$内分别代表两个解析函数，显然在这两个区域的内它们均不相等, 所以不互为解析延拓.
    \end{solution}
    
    \begin{problem}{3}
        证明$\frac{1}{z^2}$是函数$f(z) = \sum_{n=0}^{\infty}(n+1)(z+1)^n$由区域$|z+1|<1$向外的解析延拓
    \end{problem}
    \begin{solution}
        只需证明在区域$|z+1|<1$内$f(z) = \frac{1}{z^2}$即可.
        显然级数$\sum_{n=1}^{\infty}(z+1)^n$在$|z+1|<1$内闭一致收敛，可逐项求导数，所以  
        \begin{equation*}
            \begin{split}
                f(z) & = \sum_{n=0}^{\infty}(n+1)(z+1)^n\\
                &= \sum_{n=0}^{\infty}n(z+1)^n + \sum_{n=0}^{\infty}(z+1)^n\\
                &= (z+1)\sum_{n=0}^{\infty}n(z+1)^{n-1} + \sum_{n=0}^{\infty}(z+1)^n\\
                &= (z+1)\left(\sum_{n=0}^{\infty}(z+1)^n\right)^{'} + \sum_{n=0}^{\infty}(z+1)^n\\
                &= (z+1)\frac{1}{z^2} - \frac{1}{z}\\
                & = \frac{1}{z^2}
            \end{split}
        \end{equation*}
    \end{solution}
